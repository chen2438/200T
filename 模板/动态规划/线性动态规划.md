#### 代码

```cpp
#include<bits/stdc++.h>
#define FOR(i,a,b) for(int i=(a);i<=(b);++i)
using namespace std;

const int N = 510, INF = 1e9;

int n;
int a[N][N];
int f[N][N];

int main(){
    cin>>n;
    FOR(i,1,n) FOR(j,1,i) cin>>a[i][j];
    memset(f,-0x3f,sizeof(f));
    f[1][1] = a[1][1];
    FOR(i,2,n)
        FOR(j,1,i)
            f[i][j] = max(f[i - 1][j - 1] + a[i][j], f[i - 1][j] + a[i][j]);
    int res = -INF;
    FOR(i,1,n) res = max(res, f[n][i]);
    cout<<res<<endl;
    return 0;
}
```
