# 图的存储和遍历

```cpp
struct Edge{
	int to,nxt;
}e[M];

int adt,head[N];

void add(int u,int v){
	e[++adt]={v,head[u]};
	head[u]=adt;
}

int fa[N];

void dfs(int p1){
	for(int i=head[p1];i!=0;i=e[i].nxt){
		int p2=e[i].to;
		if(p2==fa[p1]) continue;
		fa[p2]=p1;
		dfs(p2);
	}
}

void bfs(int root){
	queue<int> q;
	q.push(root);
	while(!q.empty()){
		auto p1=q.front(); q.pop();
		for(int i=head[p1];i!=0;i=e[i].nxt){
			int p2=e[i].to;
			if(p2==fa[p1]) continue;
			fa[p2]=p1;
			q.push(p2);
		}
	}
}
```
