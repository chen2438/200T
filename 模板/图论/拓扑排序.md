### 代码

有向无环图一定有拓扑序，有向有环图一定没有拓扑序。

```cpp
#include<bits/stdc++.h>
#define FOR(i,a,b) for(int i=(a);i<=(b);++i)
using namespace std;

const int N = 100010, M = N;

int n, m;

struct Edge{
	int to,nxt;
}e[M];

int adt,head[N],d[N];

void add(int u,int v){
	e[++adt]={v,head[u]};
	head[u]=adt;
}

vector<int> topsort(){
    vector<int> ans;
    queue<int> q;
    FOR(i,1,n)
        if(!d[i]) q.push(i);
    while (!q.empty()){
        int p1 = q.front(); q.pop();
        ans.push_back(p1);
        for (int i = head[p1]; i ; i = e[i].nxt){
            int p2 = e[i].to;
            if (-- d[p2] == 0) q.push(p2);
        }
    }
    return ans;//返回拓扑序，若size()!=n则不存在拓扑序
}

int main(){
    cin>>n>>m;
    FOR(i,1,m){
        int a, b; cin>>a>>b;
        add(a, b);
        d[b] ++ ;
    }
    vector<int> ans = topsort();
    if (ans.size()!=n) puts("-1");
    else{
        for(auto i:ans) cout<<i<<" ";
        cout<<endl;
    }
    return 0;
}
```
